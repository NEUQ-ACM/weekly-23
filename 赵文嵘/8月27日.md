class Solution {
    public int integerBreak(int n) {
        int ans=1;
        for(int i=0;i<=n/2;i++){
            for(int j=0;j*3+i*2<=n;j++){
                if(i*2+j*3==n){
                    ans=Math.max(ans,(int)Math.pow(3,j)*(int)Math.pow(2,i));
                }
            }
        }
        if(n==2){
            return 1;
        }
        else if(n==3){
            return 2;
        }
        return ans;
    }
}

class Solution {
    public List<String> summaryRanges(int[] nums) {
        List<String> ans=new ArrayList<>();
        int first=-1,last=-1;//åŒºé—´é¦–å°¾
        int l=nums.length;
        for(int i=0;i<l;i++){
            if(first==last){
                first=last=nums[i];
            }
            if(i+1<l&&nums[i]==nums[i+1]-1){
                last=nums[i+1];
            }
            else{
                if(last!=first){
                    ans.add(new String(first+""+"->"+last));
                    last=first=-1;
                }
                else{
                    ans.add(new String(last+""));
                }
            }
        }
        return ans;
    }
}
